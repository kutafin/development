CREATE TABLE shorturl(
idUrl INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
url VARCHAR(255) NOT NULL,
shortUrl VARCHAR(45) NOT NULL,
description VARCHAR(255),
views INT,
id INT,
idTag INT
)

CREATE TABLE roles(
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(45) NOT NULL
)

CREATE TABLE users(
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
username VARCHAR(45) NOT NULL,
password VARCHAR(255) NOT NULL
)

CREATE TABLE tags(
idTag INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
tagname VARCHAR(255) NOT NULL
)

CREATE TABLE user_roles(
user_id INT NOT NUll,
role_id INT NOT NULL,
role_id INDEX user_roles(role_id ASC),
user_id UNIQUE INDEX user_roles(user_id ASC, role_id ASC),
user_roles_ibfk_1 FOREIGN KEY(user_id) REFERENCES users(id),
user_roles_ibfk_2 FOREIGN KEY(role_id) REFERENCES roles(id)
)
